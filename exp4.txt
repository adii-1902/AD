import android.content.*;
import android.os.Bundle;
import android.widget.*;
import androidx.appcompat.app.AppCompatActivity;
public class MainActivity extends AppCompatActivity {
    EditText nameInput;
    Spinner subjectSpinner;
    RadioGroup genderGroup;
    CheckBox grad, pg, phd;
    @Override
    protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_main);
        nameInput = findViewById(R.id.nameInput);
        subjectSpinner = findViewById(R.id.subjectSpinner);
        genderGroup = findViewById(R.id.genderGroup);
        grad = findViewById(R.id.grad); pg = findViewById(R.id.pg); phd = findViewById(R.id.phd);
        String[] subs = {"CS", "IT", "ENTC"};
        subjectSpinner.setAdapter(new ArrayAdapter<>(this, android.R.layout.simple_spinner_dropdown_item, subs));
        findViewById(R.id.submit).setOnClickListener(v -> {
            String name = nameInput.getText().toString();
            if (name.isEmpty() || genderGroup.getCheckedRadioButtonId() == -1) {
                Toast.makeText(this, "Fill all fields", Toast.LENGTH_SHORT).show();
                return;
            }
            String gender = ((RadioButton) findViewById(genderGroup.getCheckedRadioButtonId())).getText().toString();
            String q = "";
            if (grad.isChecked()) q += "Graduate ";
            if (pg.isChecked()) q += "Post Grad ";
            if (phd.isChecked()) q += "PhD";
            SharedPreferences.Editor e = getSharedPreferences("data", MODE_PRIVATE).edit();
            e.putString("name", name);
            e.putString("branch", subjectSpinner.getSelectedItem().toString());
            e.putString("gender", gender);
            e.putString("qual", q.trim());
            e.apply();
            startActivity(new Intent(this, ShowActivity.class));
        });
    }
}

import android.content.*;
import android.os.Bundle;
import android.widget.*;
import androidx.appcompat.app.AppCompatActivity;
public class ShowActivity extends AppCompatActivity {
    @Override
    protected void onCreate(Bundle b) {
        super.onCreate(b);
        setContentView(R.layout.activity_show);
        SharedPreferences p = getSharedPreferences("data", MODE_PRIVATE);
        String result = "Name: " + p.getString("name", "") +
                "\nBranch: " + p.getString("subject", "") +
                "\nGender: " + p.getString("gender", "") +
                "\nQualifications: " + p.getString("qual", "None");
        ((TextView) findViewById(R.id.result)).setText(result);
        findViewById(R.id.back).setOnClickListener(v -> startActivity(new Intent(this, MainActivity.class)));
    }
}
